import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

public class PessoaTest {
    @Test
    public void testGetName() {
        Pessoa pessoa = new Pessoa("John Doe");
        assertEquals("John Doe", pessoa.getName());
    }
}

public class JogadorTest {
    @Test
    public void testGetScore() {
        Jogador jogador = new Jogador("John Doe", 100, 500.0);
        assertEquals(100, jogador.getScore());
    }

    @Test
    public void testGetMoney() {
        Jogador jogador = new Jogador("John Doe", 100, 500.0);
        assertEquals(500.0, jogador.getMoney());
    }
 @Test
    public void testSetScore() {
        Jogador jogador = new Jogador("John Doe", 100, 500.0);
        jogador.setScore(200);
        assertEquals(200, jogador.getScore());
    }

    @Test
    public void testSetMoney() {
        Jogador jogador = new Jogador("John Doe", 100, 500.0);
        jogador.setMoney(1000.0);
        assertEquals(1000.0, jogador.getMoney());
    }
}

public class PerguntaTest {
    @Test
    public void testGetQuestionText() {
        String[] options = {"Option 1", "Option 2", "Option 3"};
        Pergunta pergunta = new Pergunta("What is the capital of Brazil?", options, 0, 100.0);
        assertEquals("What is the capital of Brazil?", pergunta.getQuestionText());
    }
 @Test
    public void testGetOptions() {
        String[] options = {"Option 1", "Option 2", "Option 3"};
        Pergunta pergunta = new Pergunta("What is the capital of Brazil?", options, 0, 100.0);
        assertArrayEquals(options, pergunta.getOptions());
    }

    @Test
    public void testGetCorrectAnswerIndex() {
        String[] options = {"Option 1", "Option 2", "Option 3"};
        Pergunta pergunta = new Pergunta("What is the capital of Brazil?", options, 0, 100.0);
        assertEquals(0, pergunta.getCorrectAnswerIndex());
    }

    @Test
    public void testGetValue() {
        String[] options = {"Option 1", "Option 2", "Option 3"};
        Pergunta pergunta = new Pergunta("What is the capital of Brazil?", options, 0, 100.0);
        assertEquals(100.0, pergunta.getValue());
    }
}
